## 语法结构

1. if语句——基本逻辑分支控制
```c
if (条件) {
    符合条件执行的代码;
} else if (条件) {
    符合条件执行的代码;
} else {
    条件都不符合的时候执行代码;
}
```  

2. switch语句——分支控制
```c
switch (常量) {
    case 如果常量是这个值:
        执行这部分代码;
        break;
    default:
        都不符合执行这部分代码;
}
```  

3. while循环
```c
while (条件) {
    符合条件执行代码;
}
```  

```c
while (条件) {
    if (追加的条件) {
        满足追加条件执行;
        continue; //跳过后续的代码返回循环的头部执行
    }
    执行代码;
}
```  

```c
while (条件) {
    if (追加条件) {
        满足追加条件执行代码;
        break;//退出while循环
    }
    执行代码;
}
```  

```c
do {
    执行代码;
} while(条件); 
/*
**判断条件是否满足,满足继续执行循环
**不满足执行后续语句
**/
```  

```c
for (初始条件;条件判断;条件变更) {
    执行代码;
}
```  

```c
enum{常量1, 常量2, 常量3 }常量名称;
```  

```c
if (条件) {
    goto 标签; //符合条件时跳转到标签位置代码处执行
}
标签:
    执行代码;
```  

```c
函数返回值类型 函数名称(参数1, 参数2, 参数3……) {
    执行代码;
    return 返回值;
}
```  

```c
typedef 原类型的名称 新类型的名称;
```  

```c
struct 结构体名称 {
    包含的数据类型;
}[别名];
```  

```c
typedef struct {
    包含的数据类型;
} 数据类型的名称;
```  

```c
union 联合的名称 {
    数据类型;
}[别名];
```
